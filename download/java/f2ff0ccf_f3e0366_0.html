  1. public class TestFor05{
  2.         public static void main(String[] args){
  3.                 //功能：输出1-100中被6整除的数：
  4.                 //方式1：
  5.                 /*

  11.                 */
  12.                 
  13.                 //方式2：
  14.                 for(int i=1;i<=100;i++){

  20.         }
  21. }
