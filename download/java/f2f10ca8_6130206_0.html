【1】通过练习感受break的作用：作用：停止循环： 
  1. public class TestFor02{
  2.         public static void main(String[] args){
  3.                 //功能：求1-100的和，当和第一次超过300的时候，停止程序
  4.                 int sum = 0;
  5.                 for(int i=1;i<=100;i++){	
  6.                         sum += i;	
  7.                         if(sum>300){//当和第一次超过300的时候
  8.                                 //停止循环
  9.                                 break;//停止循环
  10.                         }
  11.                         System.out.println(sum);
  12.                 }
  13.                 
  14.         }
  15. }
