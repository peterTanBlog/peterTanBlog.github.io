指定查询条件使用where子句，可以查询符合条件的部分记录。 

  1. -- 查看emp表：
  2. select * from emp;
  4. -- where子句：将过滤条件放在where子句的后面，可以筛选/过滤出我们想要的符合条件的数据：
  5. -- where 子句 + 关系运算符
  6. select * from emp where deptno = 10;
  7. select * from emp where deptno > 10;
  8. select * from emp where deptno >= 10;
  9. select * from emp where deptno < 10;
  10. select * from emp where deptno <= 10;
  11. select * from emp where deptno <> 10;
  12. select * from emp where deptno != 10;
  13. select * from emp where job = 'CLERK'; 
  14. select * from emp where job = 'clerk'; -- 默认情况下不区分大小写 
  15. select * from emp where binary job = 'clerk'; -- binary区分大小写
  16. select * from emp where hiredate < '1981-12-25';
  18. -- where 子句 + 逻辑运算符：and 
  20. select * from emp where sal > 1500 and sal < 3000;  -- (1500,3000)
  21. select * from emp where sal > 1500 && sal < 3000; 
  22. select * from emp where sal > 1500 and sal < 3000 order by sal;
  23. select * from emp where sal between 1500 and 3000; -- [1500,3000]
  26. -- where 子句 + 逻辑运算符：or
  28. select * from emp where deptno = 10 or deptno = 20;
  29. select * from emp where deptno = 10 || deptno = 20;
  30. select * from emp where deptno in (10,20);
  31. select * from emp where job in ('MANAGER','CLERK','ANALYST');
  34. -- where子句 + 模糊查询：
  35. -- 查询名字中带A的员工  -- %代表任意多个字符 0,1,2，.....
  36. select * from emp where ename like '%A%' ;
  37. -- -任意一个字符
  38. select * from emp where ename like '__A%' ;
  41. -- 关于null的判断：
  42. select * from emp where comm is null;
  43. select * from emp where comm is not null;
  46. -- 小括号的使用  ：因为不同的运算符的优先级别不同，加括号为了可读性
  47. select * from emp where job = 'SALESMAN' or job = 'CLERK' and sal >=1500; -- 先and再or  and > or
  48. select * from emp where job = 'SALESMAN' or (job = 'CLERK' and sal >=1500); 
  49. select * from emp where (job = 'SALESMAN' or job = 'CLERK') and sal >=1500;
